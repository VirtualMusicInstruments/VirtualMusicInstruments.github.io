# Python: Simple Sound Generator
import numpy as np
import sounddevice as sd

fs = 44100  # Sample rate
t = np.linspace(0, 1, fs)
wave = np.sin(2 * np.pi * 440 * t)

sd.play(wave, samplerate=fs)
sd.wait()
